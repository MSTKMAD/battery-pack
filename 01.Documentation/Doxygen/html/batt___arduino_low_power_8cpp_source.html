5static void configGCLK6()
6{
7 // enable EIC clock
8 GCLK->CLKCTRL.bit.CLKEN = 0; //disable GCLK module
9 while (GCLK->STATUS.bit.SYNCBUSY)
10 ;
11
12 GCLK->CLKCTRL.reg = (uint16_t)(GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK6 | GCLK_CLKCTRL_ID(GCM_EIC)); //EIC clock switched on GCLK6
13 while (GCLK->STATUS.bit.SYNCBUSY)
14 ;
15
16 GCLK->GENCTRL.reg = (GCLK_GENCTRL_GENEN | GCLK_GENCTRL_SRC_OSCULP32K | GCLK_GENCTRL_ID(6)); //source for GCLK6 is OSCULP32K
17 while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY)
18 ;
