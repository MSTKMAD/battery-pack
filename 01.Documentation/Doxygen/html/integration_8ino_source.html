609void setupBQ27441()
610{
611 if (!lipo.begin()) // begin() will return true if communication is successful
612 {
613 // If communication fails, print an error message and loop forever.
614 Serial.println("Error: Unable to communicate with BQ27441.");
615 Serial.println(" Check wiring and try again.");
616 Serial.println(" (Battery must be plugged into Battery Babysitter!)");
617 while (1)
618 ;
619 }
620 Serial.println("Connected to BQ27441!");
621 Serial.println("Configure?(Y/N)");
622 while (!Serial.available() && ((millis() - timer_wd) < C_TIMER_WD))
623 {
624 // Serial.print((millis() - timer_wd));
625 // Serial.print("/");
626 // Serial.println(C_TIMER_WD);
627 }
628 if (Serial.available())
629 {
630 char c = toupper(Serial.read());
631 if (c == 'Y')
632 {
633 lipo.enterConfig();
634 lipo.setCapacity(BATTERY_CAPACITY);
635 lipo.setDesignEnergy(DESIGNE_ENERGY);
636 lipo.setTerminateVoltage(TERMITANTE_VOLTAGE);
637 lipo.setTaperRate(TAPER_RATE);
638
639 lipo.exitConfig();
640 }
641 while (Serial.available())
642 {
643 Serial.read();
644 }
645 }
