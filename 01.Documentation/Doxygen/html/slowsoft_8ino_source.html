2/* Write MEMLEN bytes of EEPROM and then read it back.

26void loop() {

30 if (!si.i2c_write(0)) error();

36 if (!si.i2c_write(0)) error();

39 for (i=0; i < MEMLEN-1; i++)
40 if (si.i2c_read(false) != 0xA1) error();
41 if (si.i2c_read(true) != 0xA1) error();
42 si.i2c_stop();
