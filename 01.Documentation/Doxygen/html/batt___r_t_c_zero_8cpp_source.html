42void RTCZero::begin(bool resetTime)
43{
44 uint16_t tmp_reg = 0;
45
46 PM->APBAMASK.reg |= PM_APBAMASK_RTC; // turn on digital interface clock
47 config32kOSC();
48
49 // If the RTC is in clock mode and the reset was
50 // not due to POR or BOD, preserve the clock time
51 // POR causes a reset anyway, BOD behaviour is?
52 bool validTime = false;
53 RTC_MODE2_CLOCK_Type oldTime;
54
55 if ((!resetTime) && (PM->RCAUSE.reg & (PM_RCAUSE_SYST | PM_RCAUSE_WDT | PM_RCAUSE_EXT))) {
56 if (RTC->MODE2.CTRL.reg & RTC_MODE2_CTRL_MODE_CLOCK) {
57 validTime = true;
58 oldTime.reg = RTC->MODE2.CLOCK.reg;
59 }
60 }
61
62 // Setup clock GCLK2 with OSC32K divided by 32
63 configureClock();
64
65 RTCdisable();
66
67 RTCreset();
68
69 tmp_reg |= RTC_MODE2_CTRL_MODE_CLOCK; // set clock operating mode
70 tmp_reg |= RTC_MODE2_CTRL_PRESCALER_DIV1024; // set prescaler to 1024 for MODE2
71 tmp_reg &= ~RTC_MODE2_CTRL_MATCHCLR; // disable clear on match
72
73 //According to the datasheet RTC_MODE2_CTRL_CLKREP = 0 for 24h
74 tmp_reg &= ~RTC_MODE2_CTRL_CLKREP; // 24h time representation
75
76 RTC->MODE2.READREQ.reg &= ~RTC_READREQ_RCONT; // disable continuously mode
77
78 RTC->MODE2.CTRL.reg = tmp_reg;
79 while (RTCisSyncing())
80 ;
81
82 NVIC_EnableIRQ(RTC_IRQn); // enable RTC interrupt
83 NVIC_SetPriority(RTC_IRQn, 0x00);
84
85 RTC->MODE2.INTENSET.reg |= RTC_MODE2_INTENSET_ALARM0; // enable alarm interrupt
86 RTC->MODE2.Mode2Alarm[0].MASK.bit.SEL = MATCH_OFF; // default alarm match is off (disabled)
87
88 while (RTCisSyncing())
89 ;
90
91 RTCenable();
92 RTCresetRemove();
93
94 // If desired and valid, restore the time value, else use first valid time value
95 if ((!resetTime) && (validTime) && (oldTime.reg != 0L)) {
96 RTC->MODE2.CLOCK.reg = oldTime.reg;
97 }
98 else {
99 RTC->MODE2.CLOCK.reg = RTC_MODE2_CLOCK_YEAR(DEFAULT_YEAR - 2000) | RTC_MODE2_CLOCK_MONTH(DEFAULT_MONTH)
100 | RTC_MODE2_CLOCK_DAY(DEFAULT_DAY) | RTC_MODE2_CLOCK_HOUR(DEFAULT_HOUR)
101 | RTC_MODE2_CLOCK_MINUTE(DEFAULT_MINUTE) | RTC_MODE2_CLOCK_SECOND(DEFAULT_SECOND);
102 }
103 while (RTCisSyncing())
104 ;
105
106 _configured = true;
107}

373uint32_t RTCZero::getEpoch()
374{
375 RTCreadRequest();
376 RTC_MODE2_CLOCK_Type clockTime;
377 clockTime.reg = RTC->MODE2.CLOCK.reg;
378
379 struct tm tm;
380
381 tm.tm_isdst = -1;
382 tm.tm_yday = 0;
383 tm.tm_wday = 0;
384 tm.tm_year = clockTime.bit.YEAR + EPOCH_TIME_YEAR_OFF;
385 tm.tm_mon = clockTime.bit.MONTH - 1;
386 tm.tm_mday = clockTime.bit.DAY;
387 tm.tm_hour = clockTime.bit.HOUR;
388 tm.tm_min = clockTime.bit.MINUTE;
389 tm.tm_sec = clockTime.bit.SECOND;
390
391 return mktime(&tm);
392}

399void RTCZero::setAlarmEpoch(uint32_t ts)
400{
401 if (_configured) {
402 if (ts < EPOCH_TIME_OFF) {
403 ts = EPOCH_TIME_OFF;
404 }
405
406 time_t t = ts;
407 struct tm* tmp = gmtime(&t);
408
409 setAlarmDate(tmp->tm_mday, tmp->tm_mon + 1, tmp->tm_year - EPOCH_TIME_YEAR_OFF);
410 setAlarmTime(tmp->tm_hour, tmp->tm_min, tmp->tm_sec);
411 }

414void RTCZero::setEpoch(uint32_t ts)
415{
416 if (_configured) {
417 if (ts < EPOCH_TIME_OFF) {
418 ts = EPOCH_TIME_OFF;
419 }
420
421 time_t t = ts;
422 struct tm* tmp = gmtime(&t);
423
424 RTC_MODE2_CLOCK_Type clockTime;
425
426 clockTime.bit.YEAR = tmp->tm_year - EPOCH_TIME_YEAR_OFF;
427 clockTime.bit.MONTH = tmp->tm_mon + 1;
428 clockTime.bit.DAY = tmp->tm_mday;
429 clockTime.bit.HOUR = tmp->tm_hour;
430 clockTime.bit.MINUTE = tmp->tm_min;
431 clockTime.bit.SECOND = tmp->tm_sec;
432
433 RTC->MODE2.CLOCK.reg = clockTime.reg;
434
435 while (RTCisSyncing())
436 ;
437 }

447/* Attach peripheral clock to 32k oscillator */
448void RTCZero::configureClock() {
449 GCLK->GENDIV.reg = GCLK_GENDIV_ID(2)|GCLK_GENDIV_DIV(4);
450 while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY)
451 ;
452#ifdef CRYSTALLESS
453 GCLK->GENCTRL.reg = (GCLK_GENCTRL_GENEN | GCLK_GENCTRL_SRC_OSCULP32K | GCLK_GENCTRL_ID(2) | GCLK_GENCTRL_DIVSEL );
454#else
455 GCLK->GENCTRL.reg = (GCLK_GENCTRL_GENEN | GCLK_GENCTRL_SRC_XOSC32K | GCLK_GENCTRL_ID(2) | GCLK_GENCTRL_DIVSEL );
456#endif
457 while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY)
458 ;
459 GCLK->CLKCTRL.reg = (uint32_t)((GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK2 | (RTC_GCLK_ID << GCLK_CLKCTRL_ID_Pos)));
460 while (GCLK->STATUS.bit.SYNCBUSY)
461 ;
462}
