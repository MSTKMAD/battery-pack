160void Adafruit_MonoOLED::oled_command(uint8_t c) {
161 if (i2c_dev) { // I2C
162 uint8_t buf[2] = {0x00, c}; // Co = 0, D/C = 0
163 i2c_dev->write(buf, 2);
164 } else { // SPI (hw or soft) -- transaction started in calling function
165 digitalWrite(dcPin, LOW);
166 spi_dev->write(&c, 1);
167 }
168}

180 if (i2c_dev) { // I2C

183 return false;
184 }
185 } else { // SPI -- transaction started in calling function
186 digitalWrite(dcPin, LOW);
187 if (!spi_dev->write((uint8_t *)c, n)) {
188 return false;
189 }
190 }
191 return true;
192}

216bool Adafruit_MonoOLED::_init(uint8_t addr, bool reset) {
217
218 // attempt to malloc the bitmap framebuffer
219 if ((!buffer) && !(buffer = (uint8_t *)malloc(WIDTH * ((HEIGHT + 7) / 8)))) {
220 return false;
221 }
222
223 // Reset OLED if requested and reset pin specified in constructor
224 if (reset && (rstPin >= 0)) {
225 pinMode(rstPin, OUTPUT);
226 digitalWrite(rstPin, HIGH);
227 delay(10); // VDD goes high at start, pause
228 digitalWrite(rstPin, LOW); // Bring reset low
229 delay(10); // Wait 10 ms
230 digitalWrite(rstPin, HIGH); // Bring out of reset
231 delay(10);
232 }
233
234 // Setup pin directions
235 if (_theWire) { // using I2C
236 i2c_dev = new Adafruit_I2CDevice(addr, _theWire);
237 // look for i2c address:
238 if (!i2c_dev || !i2c_dev->begin()) {
239 return false;
240 }
241 } else { // Using one of the SPI modes, either soft or hardware
242 if (!spi_dev || !spi_dev->begin()) {
243 return false;
244 }
245 pinMode(dcPin, OUTPUT); // Set data/command pin as output
246 }

251 window_x1 = 0;

277void Adafruit_MonoOLED::drawPixel(int16_t x, int16_t y, uint16_t color) {
278 if ((x >= 0) && (x < width()) && (y >= 0) && (y < height())) {
279 // Pixel is in-bounds. Rotate coordinates if needed.
280 switch (getRotation()) {
281 case 1:
282 monooled_swap(x, y);
283 x = WIDTH - x - 1;
284 break;
285 case 2:
286 x = WIDTH - x - 1;
287 y = HEIGHT - y - 1;
288 break;
289 case 3:
290 monooled_swap(x, y);
291 y = HEIGHT - y - 1;
292 break;
293 }
294
295 // adjust dirty window
296 window_x1 = min(window_x1, x);
297 window_y1 = min(window_y1, y);
298 window_x2 = max(window_x2, x);
299 window_y2 = max(window_y2, y);
300
301 switch (color) {
302 case MONOOLED_WHITE:
303 buffer[x + (y / 8) * WIDTH] |= (1 << (y & 7));
304 break;
305 case MONOOLED_BLACK:
306 buffer[x + (y / 8) * WIDTH] &= ~(1 << (y & 7));
307 break;
308 case MONOOLED_INVERSE:
309 buffer[x + (y / 8) * WIDTH] ^= (1 << (y & 7));
310 break;
311 }
312 }
313}

342bool Adafruit_MonoOLED::getPixel(int16_t x, int16_t y) {
343 if ((x >= 0) && (x < width()) && (y >= 0) && (y < height())) {
344 // Pixel is in-bounds. Rotate coordinates if needed.
345 switch (getRotation()) {
346 case 1:
347 monooled_swap(x, y);

349 break;

352 y = HEIGHT - y - 1;
353 break;
354 case 3:

356 y = HEIGHT - y - 1;
357 break;
358 }
