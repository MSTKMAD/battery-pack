2/* Write MEMLEN bytes of EEPROM and then read it back.
3 * If successful, light a LED with slow blinks, otherwise blink very hecticly.
4 * The baseline uses almost empty functions, the other
5 * implementations are measured against it
6 */

8#define MEMADDR7B 0x57 // 7-bit addr of memory chip
9#define DATA 0xA1 // can be changed

11#define MEMLEN 10 // the number of bytes to be written and to be read

25void loop() {
26 byte i;
27 // writing 10 bytes
28 Wire.beginTransmission(MEMADDR7B);
29 for (i=0; i < ADDRLEN; i++)
30 Wire.write(0);
31 for (i=0; i<MEMLEN; i++)
32 Wire.write(DATA);
33 if (Wire.endTransmission() != 0) error();
34
35 // setting register addres (waiting for an ACK)
36
37 while (true) {
38 Wire.beginTransmission(MEMADDR7B);
39 for (i=0; i < ADDRLEN; i++)
40 Wire.write(0);
41 if (Wire.endTransmission() == 0) break;
42 }
43 if (Wire.requestFrom(MEMADDR7B, MEMLEN) != MEMLEN) error();
44 i = 0;
45 while (Wire.available()) {
46 if (Wire.read() != DATA) error();
47 }

49 delay(2000);
50 digitalWrite(LEDPIN, LOW);
51 delay(2000);
52}
