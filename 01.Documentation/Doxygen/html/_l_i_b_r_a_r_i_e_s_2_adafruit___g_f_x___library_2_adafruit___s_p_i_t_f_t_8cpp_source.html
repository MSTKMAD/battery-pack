527void Adafruit_SPITFT::initSPI(uint32_t freq, uint8_t spiMode) {
528
529 if (!freq)
530 freq = DEFAULT_SPI_FREQ; // If no freq specified, use default
531
532 // Init basic control pins common to all connection types
533 if (_cs >= 0) {
534 pinMode(_cs, OUTPUT);
535 digitalWrite(_cs, HIGH); // Deselect
536 }
537 pinMode(_dc, OUTPUT);
538 digitalWrite(_dc, HIGH); // Data mode
539
540 if (connection == TFT_HARD_SPI) {
541
542#if defined(SPI_HAS_TRANSACTION)
543 hwspi.settings = SPISettings(freq, MSBFIRST, spiMode);
544#else
545 hwspi._freq = freq; // Save freq value for later
546#endif
547 hwspi._mode = spiMode; // Save spiMode value for later
548 // Call hwspi._spi->begin() ONLY if this is among the 'established'
549 // SPI interfaces in variant.h. For DIY roll-your-own SERCOM SPIs,
550 // begin() and pinPeripheral() calls MUST be made in one's calling
551 // code, BEFORE the screen-specific begin/init function is called.
552 // Reason for this is that SPI::begin() makes its own calls to
553 // pinPeripheral() based on g_APinDescription[n].ulPinType, which
554 // on non-established SPI interface pins will always be PIO_DIGITAL
555 // or similar, while we need PIO_SERCOM or PIO_SERCOM_ALT...it's
556 // highly unique between devices and variants for each pin or
557 // SERCOM so we can't make those calls ourselves here. And the SPI
558 // device needs to be set up before calling this because it's
559 // immediately followed with initialization commands. Blargh.
560 if (
561#if !defined(SPI_INTERFACES_COUNT)
562 1
563#else
564#if SPI_INTERFACES_COUNT > 0
565 (hwspi._spi == &SPI)
566#endif
567#if SPI_INTERFACES_COUNT > 1
568 || (hwspi._spi == &SPI1)
569#endif
570#if SPI_INTERFACES_COUNT > 2
571 || (hwspi._spi == &SPI2)
572#endif
573#if SPI_INTERFACES_COUNT > 3
574 || (hwspi._spi == &SPI3)
575#endif
576#if SPI_INTERFACES_COUNT > 4
577 || (hwspi._spi == &SPI4)
578#endif
579#if SPI_INTERFACES_COUNT > 5
580 || (hwspi._spi == &SPI5)
581#endif
582#endif // end SPI_INTERFACES_COUNT
583 ) {
584 hwspi._spi->begin();
585 }
586 } else if (connection == TFT_SOFT_SPI) {
587
588 pinMode(swspi._mosi, OUTPUT);
589 digitalWrite(swspi._mosi, LOW);
590 pinMode(swspi._sck, OUTPUT);
591 digitalWrite(swspi._sck, LOW);
592 if (swspi._miso >= 0) {
593 pinMode(swspi._miso, INPUT);
594 }
595
596 } else { // TFT_PARALLEL
597 // Initialize data pins. We were only passed d0, so scan
598 // the pin description list looking for the other pins.
599 // They'll be on the same PORT, and within the next 7 (or 15) bits
600 // (because we need to write to a contiguous PORT byte or word).
601#if defined(__AVR__)
602 // PORT registers are 8 bits wide, so just need a register match...
603 for (uint8_t i = 0; i < NUM_DIGITAL_PINS; i++) {
604 if ((PORTreg_t)portOutputRegister(digitalPinToPort(i)) ==
605 tft8.writePort) {
606 pinMode(i, OUTPUT);
607 digitalWrite(i, LOW);
608 }
609 }
610#elif defined(USE_FAST_PINIO)
611#if defined(CORE_TEENSY)
612 if (!tft8.wide) {
613 *tft8.dirSet = 0xFF; // Set port to output
614 *tft8.writePort = 0x00; // Write all 0s
615 } else {
616 *(volatile uint16_t *)tft8.dirSet = 0xFFFF;
617 *(volatile uint16_t *)tft8.writePort = 0x0000;
618 }
619#else // !CORE_TEENSY
620 uint8_t portNum = g_APinDescription[tft8._d0].ulPort, // d0 PORT #
621 dBit = g_APinDescription[tft8._d0].ulPin, // d0 bit in PORT
622 lastBit = dBit + (tft8.wide ? 15 : 7);
623 for (uint8_t i = 0; i < PINS_COUNT; i++) {
624 if ((g_APinDescription[i].ulPort == portNum) &&
625 (g_APinDescription[i].ulPin >= dBit) &&
626 (g_APinDescription[i].ulPin <= (uint32_t)lastBit)) {
627 pinMode(i, OUTPUT);
628 digitalWrite(i, LOW);
629 }
630 }
631#endif // end !CORE_TEENSY
632#endif
633 pinMode(tft8._wr, OUTPUT);
634 digitalWrite(tft8._wr, HIGH);
635 if (tft8._rd >= 0) {
636 pinMode(tft8._rd, OUTPUT);
637 digitalWrite(tft8._rd, HIGH);
638 }
639 }
640
641 if (_rst >= 0) {
642 // Toggle _rst low to reset
643 pinMode(_rst, OUTPUT);
644 digitalWrite(_rst, HIGH);
645 delay(100);
646 digitalWrite(_rst, LOW);
647 delay(100);
648 digitalWrite(_rst, HIGH);
649 delay(200);
650 }
651
652#if defined(USE_SPI_DMA) && (defined(__SAMD51__) || defined(ARDUINO_SAMD_ZERO))
653 if (((connection == TFT_HARD_SPI) || (connection == TFT_PARALLEL)) &&
654 (dma.allocate() == DMA_STATUS_OK)) { // Allocate channel
655 // The DMA library needs to alloc at least one valid descriptor,
656 // so we do that here. It's not used in the usual sense though,
657 // just before a transfer we copy descriptor[0] to this address.
658 if (dptr = dma.addDescriptor(NULL, NULL, 42, DMA_BEAT_SIZE_BYTE, false,
659 false)) {
660 // Alloc 2 scanlines worth of pixels on display's major axis,
661 // whichever that is, rounding each up to 2-pixel boundary.
662 int major = (WIDTH > HEIGHT) ? WIDTH : HEIGHT;
663 major += (major & 1); // -> next 2-pixel bound, if needed.
664 maxFillLen = major * 2; // 2 scanlines
665 // Note to future self: if you decide to make the pixel buffer
666 // much larger, remember that DMA transfer descriptors can't
667 // exceed 65,535 bytes (not 65,536), meaning 32,767 pixels max.
668 // Not that we have that kind of RAM to throw around right now.
669 if ((pixelBuf[0] = (uint16_t *)malloc(maxFillLen * sizeof(uint16_t)))) {
670 // Alloc OK. Get pointer to start of second scanline.
671 pixelBuf[1] = &pixelBuf[0][major];
672 // Determine number of DMA descriptors needed to cover
673 // entire screen when entire 2-line pixelBuf is used
674 // (round up for fractional last descriptor).
675 int numDescriptors = (WIDTH * HEIGHT + (maxFillLen - 1)) / maxFillLen;
676 // DMA descriptors MUST be 128-bit (16 byte) aligned.
677 // memalign() is considered obsolete but it's replacements
678 // (aligned_alloc() or posix_memalign()) are not currently
679 // available in the version of ARM GCC in use, but this
680 // is, so here we are.
681 if ((descriptor = (DmacDescriptor *)memalign(
682 16, numDescriptors * sizeof(DmacDescriptor)))) {
683 int dmac_id;
684 volatile uint32_t *data_reg;
685
686 if (connection == TFT_HARD_SPI) {
687 // THIS IS AN AFFRONT TO NATURE, but I don't know
688 // any "clean" way to get the sercom number from the
689 // the SPIClass pointer (e.g. &SPI or &SPI1), which
690 // is all we have to work with. SPIClass does contain
691 // a SERCOM pointer but it is a PRIVATE member!
692 // Doing an UNSPEAKABLY HORRIBLE THING here, directly
693 // accessing the first 32-bit value in the SPIClass
694 // structure, knowing that's (currently) where the
695 // SERCOM pointer lives, but this ENTIRELY DEPENDS on
696 // that structure not changing nor the compiler
697 // rearranging things. Oh the humanity!
698
699 if (*(SERCOM **)hwspi._spi == &sercom0) {
700 dmac_id = SERCOM0_DMAC_ID_TX;
701 data_reg = &SERCOM0->SPI.DATA.reg;
702#if defined SERCOM1
703 } else if (*(SERCOM **)hwspi._spi == &sercom1) {
704 dmac_id = SERCOM1_DMAC_ID_TX;
705 data_reg = &SERCOM1->SPI.DATA.reg;
706#endif
707#if defined SERCOM2
708 } else if (*(SERCOM **)hwspi._spi == &sercom2) {
709 dmac_id = SERCOM2_DMAC_ID_TX;
710 data_reg = &SERCOM2->SPI.DATA.reg;
711#endif
712#if defined SERCOM3
713 } else if (*(SERCOM **)hwspi._spi == &sercom3) {
714 dmac_id = SERCOM3_DMAC_ID_TX;
715 data_reg = &SERCOM3->SPI.DATA.reg;
716#endif
717#if defined SERCOM4
718 } else if (*(SERCOM **)hwspi._spi == &sercom4) {
719 dmac_id = SERCOM4_DMAC_ID_TX;
720 data_reg = &SERCOM4->SPI.DATA.reg;
721#endif
722#if defined SERCOM5
723 } else if (*(SERCOM **)hwspi._spi == &sercom5) {
724 dmac_id = SERCOM5_DMAC_ID_TX;
725 data_reg = &SERCOM5->SPI.DATA.reg;
726#endif
727#if defined SERCOM6
728 } else if (*(SERCOM **)hwspi._spi == &sercom6) {
729 dmac_id = SERCOM6_DMAC_ID_TX;
730 data_reg = &SERCOM6->SPI.DATA.reg;
731#endif
732#if defined SERCOM7
733 } else if (*(SERCOM **)hwspi._spi == &sercom7) {
734 dmac_id = SERCOM7_DMAC_ID_TX;
735 data_reg = &SERCOM7->SPI.DATA.reg;
736#endif
737 }
738 dma.setPriority(DMA_PRIORITY_3);
739 dma.setTrigger(dmac_id);
740 dma.setAction(DMA_TRIGGER_ACTON_BEAT);
741
742 // Initialize descriptor list.
743 for (int d = 0; d < numDescriptors; d++) {
744 // No need to set SRCADDR, DESCADDR or BTCNT --
745 // those are done in the pixel-writing functions.
746 descriptor[d].BTCTRL.bit.VALID = true;
747 descriptor[d].BTCTRL.bit.EVOSEL = DMA_EVENT_OUTPUT_DISABLE;
748 descriptor[d].BTCTRL.bit.BLOCKACT = DMA_BLOCK_ACTION_NOACT;
749 descriptor[d].BTCTRL.bit.BEATSIZE = DMA_BEAT_SIZE_BYTE;
750 descriptor[d].BTCTRL.bit.DSTINC = 0;
751 descriptor[d].BTCTRL.bit.STEPSEL = DMA_STEPSEL_SRC;
752 descriptor[d].BTCTRL.bit.STEPSIZE =
753 DMA_ADDRESS_INCREMENT_STEP_SIZE_1;
754 descriptor[d].DSTADDR.reg = (uint32_t)data_reg;
755 }
756
757 } else { // Parallel connection
758
759#if defined(__SAMD51__)
760 int dmaChannel = dma.getChannel();
761 // Enable event output, use EVOSEL output
762 DMAC->Channel[dmaChannel].CHEVCTRL.bit.EVOE = 1;
763 DMAC->Channel[dmaChannel].CHEVCTRL.bit.EVOMODE = 0;
764
765 // CONFIGURE TIMER/COUNTER (for write strobe)
766
767 Tc *timer = tcList[tcNum].tc; // -> Timer struct
768 int id = tcList[tcNum].gclk; // Timer GCLK ID
769 GCLK_PCHCTRL_Type pchctrl;
770
771 // Set up timer clock source from GCLK
772 GCLK->PCHCTRL[id].bit.CHEN = 0; // Stop timer
773 while (GCLK->PCHCTRL[id].bit.CHEN)
774 ; // Wait for it
775 pchctrl.bit.GEN = GCLK_PCHCTRL_GEN_GCLK0_Val;
776 pchctrl.bit.CHEN = 1; // Enable
777 GCLK->PCHCTRL[id].reg = pchctrl.reg;
778 while (!GCLK->PCHCTRL[id].bit.CHEN)
779 ; // Wait for it
780
781 // Disable timer/counter before configuring it
782 timer->COUNT8.CTRLA.bit.ENABLE = 0;
783 while (timer->COUNT8.SYNCBUSY.bit.STATUS)
784 ;
785
786 timer->COUNT8.WAVE.bit.WAVEGEN = 2; // NPWM
787 timer->COUNT8.CTRLA.bit.MODE = 1; // 8-bit
788 timer->COUNT8.CTRLA.bit.PRESCALER = 0; // 1:1
789 while (timer->COUNT8.SYNCBUSY.bit.STATUS)
790 ;
791
792 timer->COUNT8.CTRLBCLR.bit.DIR = 1; // Count UP
793 while (timer->COUNT8.SYNCBUSY.bit.CTRLB)
794 ;
795 timer->COUNT8.CTRLBSET.bit.ONESHOT = 1; // One-shot
796 while (timer->COUNT8.SYNCBUSY.bit.CTRLB)
797 ;
798 timer->COUNT8.PER.reg = 6; // PWM top
799 while (timer->COUNT8.SYNCBUSY.bit.PER)
800 ;
801 timer->COUNT8.CC[0].reg = 2; // Compare
802 while (timer->COUNT8.SYNCBUSY.bit.CC0)
803 ;
804 // Enable async input events,
805 // event action = restart.
806 timer->COUNT8.EVCTRL.bit.TCEI = 1;
807 timer->COUNT8.EVCTRL.bit.EVACT = 1;
808
809 // Enable timer
810 timer->COUNT8.CTRLA.reg |= TC_CTRLA_ENABLE;
811 while (timer->COUNT8.SYNCBUSY.bit.STATUS)
812 ;
813
814#if (wrPeripheral == PIO_CCL)
815 // CONFIGURE CCL (inverts timer/counter output)
816
817 MCLK->APBCMASK.bit.CCL_ = 1; // Enable CCL clock
818 CCL->CTRL.bit.ENABLE = 0; // Disable to config
819 CCL->CTRL.bit.SWRST = 1; // Reset CCL registers
820 CCL->LUTCTRL[tcNum].bit.ENABLE = 0; // Disable LUT
821 CCL->LUTCTRL[tcNum].bit.FILTSEL = 0; // No filter
822 CCL->LUTCTRL[tcNum].bit.INSEL0 = 6; // TC input
823 CCL->LUTCTRL[tcNum].bit.INSEL1 = 0; // MASK
824 CCL->LUTCTRL[tcNum].bit.INSEL2 = 0; // MASK
825 CCL->LUTCTRL[tcNum].bit.TRUTH = 1; // Invert in 0
826 CCL->LUTCTRL[tcNum].bit.ENABLE = 1; // Enable LUT
827 CCL->CTRL.bit.ENABLE = 1; // Enable CCL
828#endif
829
830 // CONFIGURE EVENT SYSTEM
831
832 // Set up event system clock source from GCLK...
833 // Disable EVSYS, wait for disable
834 GCLK->PCHCTRL[EVSYS_GCLK_ID_0].bit.CHEN = 0;
835 while (GCLK->PCHCTRL[EVSYS_GCLK_ID_0].bit.CHEN)
836 ;
837 pchctrl.bit.GEN = GCLK_PCHCTRL_GEN_GCLK0_Val;
838 pchctrl.bit.CHEN = 1; // Re-enable
839 GCLK->PCHCTRL[EVSYS_GCLK_ID_0].reg = pchctrl.reg;
840 // Wait for it, then enable EVSYS clock
841 while (!GCLK->PCHCTRL[EVSYS_GCLK_ID_0].bit.CHEN)
842 ;
843 MCLK->APBBMASK.bit.EVSYS_ = 1;
844
845 // Connect Timer EVU to ch 0
846 EVSYS->USER[tcList[tcNum].evu].reg = 1;
847 // Datasheet recommends single write operation;
848 // reg instead of bit. Also datasheet: PATH bits
849 // must be zero when using async!
850 EVSYS_CHANNEL_Type ev;
851 ev.reg = 0;
852 ev.bit.PATH = 2; // Asynchronous
853 ev.bit.EVGEN = 0x22 + dmaChannel; // DMA channel 0+
854 EVSYS->Channel[0].CHANNEL.reg = ev.reg;
855
856 // Initialize descriptor list.
857 for (int d = 0; d < numDescriptors; d++) {
858 // No need to set SRCADDR, DESCADDR or BTCNT --
859 // those are done in the pixel-writing functions.
860 descriptor[d].BTCTRL.bit.VALID = true;
861 // Event strobe on beat xfer:
862 descriptor[d].BTCTRL.bit.EVOSEL = 0x3;
863 descriptor[d].BTCTRL.bit.BLOCKACT = DMA_BLOCK_ACTION_NOACT;
864 descriptor[d].BTCTRL.bit.BEATSIZE =
865 tft8.wide ? DMA_BEAT_SIZE_HWORD : DMA_BEAT_SIZE_BYTE;
866 descriptor[d].BTCTRL.bit.SRCINC = 1;
867 descriptor[d].BTCTRL.bit.DSTINC = 0;
868 descriptor[d].BTCTRL.bit.STEPSEL = DMA_STEPSEL_SRC;
869 descriptor[d].BTCTRL.bit.STEPSIZE =
870 DMA_ADDRESS_INCREMENT_STEP_SIZE_1;
871 descriptor[d].DSTADDR.reg = (uint32_t)tft8.writePort;
872 }
873#endif // __SAMD51
874 } // end parallel-specific DMA setup
875
876 lastFillColor = 0x0000;
877 lastFillLen = 0;
878 dma.setCallback(dma_callback);
879 return; // Success!
880 // else clean up any partial allocation...
881 } // end descriptor memalign()
882 free(pixelBuf[0]);
883 pixelBuf[0] = pixelBuf[1] = NULL;
884 } // end pixelBuf malloc()
885 // Don't currently have a descriptor delete function in
886 // ZeroDMA lib, but if we did, it would be called here.
887 } // end addDescriptor()
888 dma.free(); // Deallocate DMA channel
889 }
890#endif // end USE_SPI_DMA
891}

995void Adafruit_SPITFT::writePixels(uint16_t *colors, uint32_t len, bool block,
996 bool bigEndian) {
997
998 if (!len)
999 return; // Avoid 0-byte transfers
1000
1001 // avoid paramater-not-used complaints
1002 (void)block;
1003 (void)bigEndian;
1004
1005#if defined(ESP32)
1006 if (connection == TFT_HARD_SPI) {
1007 if (!bigEndian) {
1008 hwspi._spi->writePixels(colors, len * 2); // Inbuilt endian-swap
1009 } else {
1010 hwspi._spi->writeBytes((uint8_t *)colors, len * 2); // Issue bytes direct
1011 }
1012 return;
1013 }
1014#elif defined(ARDUINO_NRF52_ADAFRUIT) && \
1015 defined(NRF52840_XXAA) // Adafruit nRF52 use SPIM3 DMA at 32Mhz
1016 if (!bigEndian) {
1017 swapBytes(colors, len); // convert little-to-big endian for display
1018 }
1019 hwspi._spi->transfer(colors, NULL, 2 * len); // NULL RX to avoid overwrite
1020 if (!bigEndian) {
1021 swapBytes(colors, len); // big-to-little endian to restore pixel buffer
1022 }
1023
1024 return;
1025#elif defined(USE_SPI_DMA) && \
1026 (defined(__SAMD51__) || defined(ARDUINO_SAMD_ZERO))
1027 if ((connection == TFT_HARD_SPI) || (connection == TFT_PARALLEL)) {
1028 int maxSpan = maxFillLen / 2; // One scanline max
1029 uint8_t pixelBufIdx = 0; // Active pixel buffer number
1030#if defined(__SAMD51__)
1031 if (connection == TFT_PARALLEL) {
1032 // Switch WR pin to PWM or CCL
1033 pinPeripheral(tft8._wr, wrPeripheral);
1034 }
1035#endif // end __SAMD51__
1036 if (!bigEndian) { // Normal little-endian situation...
1037 while (len) {
1038 int count = (len < maxSpan) ? len : maxSpan;
1039
1040 // Because TFT and SAMD endianisms are different, must swap
1041 // bytes from the 'colors' array passed into a DMA working
1042 // buffer. This can take place while the prior DMA transfer
1043 // is in progress, hence the need for two pixelBufs.
1044 swapBytes(colors, count, pixelBuf[pixelBufIdx]);
1045 colors += count;
1046
1047 // The transfers themselves are relatively small, so we don't
1048 // need a long descriptor list. We just alternate between the
1049 // first two, sharing pixelBufIdx for that purpose.
1050 descriptor[pixelBufIdx].SRCADDR.reg =
1051 (uint32_t)pixelBuf[pixelBufIdx] + count * 2;
1052 descriptor[pixelBufIdx].BTCTRL.bit.SRCINC = 1;
1053 descriptor[pixelBufIdx].BTCNT.reg = count * 2;
1054 descriptor[pixelBufIdx].DESCADDR.reg = 0;
1055
1056 while (dma_busy)
1057 ; // Wait for prior line to finish
1058
1059 // Move new descriptor into place...
1060 memcpy(dptr, &descriptor[pixelBufIdx], sizeof(DmacDescriptor));
1061 dma_busy = true;
1062 dma.startJob(); // Trigger SPI DMA transfer
1063 if (connection == TFT_PARALLEL)
1064 dma.trigger();
1065 pixelBufIdx = 1 - pixelBufIdx; // Swap DMA pixel buffers
1066
1067 len -= count;
1068 }
1069 } else { // bigEndian == true
1070 // With big-endian pixel data, this can be handled as a single
1071 // DMA transfer using chained descriptors. Even full screen, this
1072 // needs only a relatively short descriptor list, each
1073 // transferring a max of 32,767 (not 32,768) pixels. The list
1074 // was allocated large enough to accommodate a full screen's
1075 // worth of data, so this won't run past the end of the list.
1076 int d, numDescriptors = (len + 32766) / 32767;
1077 for (d = 0; d < numDescriptors; d++) {
1078 int count = (len < 32767) ? len : 32767;
1079 descriptor[d].SRCADDR.reg = (uint32_t)colors + count * 2;
1080 descriptor[d].BTCTRL.bit.SRCINC = 1;
1081 descriptor[d].BTCNT.reg = count * 2;
1082 descriptor[d].DESCADDR.reg = (uint32_t)&descriptor[d + 1];
1083 len -= count;
1084 colors += count;
1085 }
1086 descriptor[d - 1].DESCADDR.reg = 0;
1087
1088 while (dma_busy)
1089 ; // Wait for prior transfer (if any) to finish
1090
1091 // Move first descriptor into place and start transfer...
1092 memcpy(dptr, &descriptor[0], sizeof(DmacDescriptor));
1093 dma_busy = true;
1094 dma.startJob(); // Trigger SPI DMA transfer
1095 if (connection == TFT_PARALLEL)
1096 dma.trigger();
1097 } // end bigEndian
1098
1099 lastFillColor = 0x0000; // pixelBuf has been sullied
1100 lastFillLen = 0;
1101 if (block) {
1102 while (dma_busy)
1103 ; // Wait for last line to complete
1104#if defined(__SAMD51__) || defined(ARDUINO_SAMD_ZERO)
1105 if (connection == TFT_HARD_SPI) {
1106 // See SAMD51/21 note in writeColor()
1107 hwspi._spi->setDataMode(hwspi._mode);
1108 } else {
1109 pinPeripheral(tft8._wr, PIO_OUTPUT); // Switch WR back to GPIO
1110 }
1111#endif // end __SAMD51__ || ARDUINO_SAMD_ZERO
1112 }
1113 return;
1114 }
1115#endif // end USE_SPI_DMA
1116
1117 // All other cases (bitbang SPI or non-DMA hard SPI or parallel),
1118 // use a loop with the normal 16-bit data write function:
1119
1120 if (!bigEndian) {
1121 while (len--) {
1122 SPI_WRITE16(*colors++);
1123 }
1124 } else {
1125 // Well this is awkward. SPI_WRITE16() was designed for little-endian
1126 // hosts and big-endian displays as that's nearly always the typical
1127 // case. If the bigEndian flag was set, data is already in display's
1128 // order...so each pixel needs byte-swapping before being issued.
1129 // Rather than having a separate big-endian SPI_WRITE16 (adding more
1130 // bloat), it's preferred if calling function is smart and only uses
1131 // bigEndian where DMA is supported. But we gotta handle this...
1132 while (len--) {
1133 SPI_WRITE16(__builtin_bswap16(*colors++));
1134 }
1135 }
1136}

1165void Adafruit_SPITFT::writeColor(uint16_t color, uint32_t len) {
1166
1167 if (!len)
1168 return; // Avoid 0-byte transfers
1169
1170 uint8_t hi = color >> 8, lo = color;
1171
1172#if defined(ESP32) // ESP32 has a special SPI pixel-writing function...
1173 if (connection == TFT_HARD_SPI) {
1174#define SPI_MAX_PIXELS_AT_ONCE 32
1175#define TMPBUF_LONGWORDS (SPI_MAX_PIXELS_AT_ONCE + 1) / 2
1176#define TMPBUF_PIXELS (TMPBUF_LONGWORDS * 2)
1177 static uint32_t temp[TMPBUF_LONGWORDS];
1178 uint32_t c32 = color * 0x00010001;
1179 uint16_t bufLen = (len < TMPBUF_PIXELS) ? len : TMPBUF_PIXELS, xferLen,
1180 fillLen;
1181 // Fill temp buffer 32 bits at a time
1182 fillLen = (bufLen + 1) / 2; // Round up to next 32-bit boundary
1183 for (uint32_t t = 0; t < fillLen; t++) {
1184 temp[t] = c32;
1185 }
1186 // Issue pixels in blocks from temp buffer
1187 while (len) { // While pixels remain
1188 xferLen = (bufLen < len) ? bufLen : len; // How many this pass?
1189 writePixels((uint16_t *)temp, xferLen);
1190 len -= xferLen;
1191 }
1192 return;
1193 }
1194#elif defined(ARDUINO_NRF52_ADAFRUIT) && \
1195 defined(NRF52840_XXAA) // Adafruit nRF52840 use SPIM3 DMA at 32Mhz
1196 // at most 2 scan lines
1197 uint32_t const pixbufcount = min(len, ((uint32_t)2 * width()));
1198 uint16_t *pixbuf = (uint16_t *)rtos_malloc(2 * pixbufcount);
1199
1200 // use SPI3 DMA if we could allocate buffer, else fall back to writing each
1201 // pixel loop below
1202 if (pixbuf) {
1203 uint16_t const swap_color = __builtin_bswap16(color);
1204
1205 // fill buffer with color
1206 for (uint32_t i = 0; i < pixbufcount; i++) {
1207 pixbuf[i] = swap_color;
1208 }
1209
1210 while (len) {
1211 uint32_t const count = min(len, pixbufcount);
1212 writePixels(pixbuf, count, true, true);
1213 len -= count;
1214 }
1215
1216 rtos_free(pixbuf);
1217 return;
1218 }
1219#else // !ESP32
1220#if defined(USE_SPI_DMA) && (defined(__SAMD51__) || defined(ARDUINO_SAMD_ZERO))
1221 if (((connection == TFT_HARD_SPI) || (connection == TFT_PARALLEL)) &&
1222 (len >= 16)) { // Don't bother with DMA on short pixel runs
1223 int i, d, numDescriptors;
1224 if (hi == lo) { // If high & low bytes are same...
1225 onePixelBuf = color;
1226 // Can do this with a relatively short descriptor list,
1227 // each transferring a max of 32,767 (not 32,768) pixels.
1228 // This won't run off the end of the allocated descriptor list,
1229 // since we're using much larger chunks per descriptor here.
1230 numDescriptors = (len + 32766) / 32767;
1231 for (d = 0; d < numDescriptors; d++) {
1232 int count = (len < 32767) ? len : 32767;
1233 descriptor[d].SRCADDR.reg = (uint32_t)&onePixelBuf;
1234 descriptor[d].BTCTRL.bit.SRCINC = 0;
1235 descriptor[d].BTCNT.reg = count * 2;
1236 descriptor[d].DESCADDR.reg = (uint32_t)&descriptor[d + 1];
1237 len -= count;
1238 }
1239 descriptor[d - 1].DESCADDR.reg = 0;
1240 } else {
1241 // If high and low bytes are distinct, it's necessary to fill
1242 // a buffer with pixel data (swapping high and low bytes because
1243 // TFT and SAMD are different endianisms) and create a longer
1244 // descriptor list pointing repeatedly to this data. We can do
1245 // this slightly faster working 2 pixels (32 bits) at a time.
1246 uint32_t *pixelPtr = (uint32_t *)pixelBuf[0],
1247 twoPixels = __builtin_bswap16(color) * 0x00010001;
1248 // We can avoid some or all of the buffer-filling if the color
1249 // is the same as last time...
1250 if (color == lastFillColor) {
1251 // If length is longer than prior instance, fill only the
1252 // additional pixels in the buffer and update lastFillLen.
1253 if (len > lastFillLen) {
1254 int fillStart = lastFillLen / 2,
1255 fillEnd = (((len < maxFillLen) ? len : maxFillLen) + 1) / 2;
1256 for (i = fillStart; i < fillEnd; i++)
1257 pixelPtr[i] = twoPixels;
1258 lastFillLen = fillEnd * 2;
1259 } // else do nothing, don't set pixels or change lastFillLen
1260 } else {
1261 int fillEnd = (((len < maxFillLen) ? len : maxFillLen) + 1) / 2;
1262 for (i = 0; i < fillEnd; i++)
1263 pixelPtr[i] = twoPixels;
1264 lastFillLen = fillEnd * 2;
1265 lastFillColor = color;
1266 }
1267
1268 numDescriptors = (len + maxFillLen - 1) / maxFillLen;
1269 for (d = 0; d < numDescriptors; d++) {
1270 int pixels = (len < maxFillLen) ? len : maxFillLen, bytes = pixels * 2;
1271 descriptor[d].SRCADDR.reg = (uint32_t)pixelPtr + bytes;
1272 descriptor[d].BTCTRL.bit.SRCINC = 1;
1273 descriptor[d].BTCNT.reg = bytes;
1274 descriptor[d].DESCADDR.reg = (uint32_t)&descriptor[d + 1];
1275 len -= pixels;
1276 }
1277 descriptor[d - 1].DESCADDR.reg = 0;
1278 }
1279 memcpy(dptr, &descriptor[0], sizeof(DmacDescriptor));
1280#if defined(__SAMD51__)
1281 if (connection == TFT_PARALLEL) {
1282 // Switch WR pin to PWM or CCL
1283 pinPeripheral(tft8._wr, wrPeripheral);
1284 }
1285#endif // end __SAMD51__
1286
1287 dma_busy = true;
1288 dma.startJob();
1289 if (connection == TFT_PARALLEL)
1290 dma.trigger();
1291 while (dma_busy)
1292 ; // Wait for completion
1293 // Unfortunately blocking is necessary. An earlier version returned
1294 // immediately and checked dma_busy on startWrite() instead, but it
1295 // turns out to be MUCH slower on many graphics operations (as when
1296 // drawing lines, pixel-by-pixel), perhaps because it's a volatile
1297 // type and doesn't cache. Working on this.
1298#if defined(__SAMD51__) || defined(ARDUINO_SAMD_ZERO)
1299 if (connection == TFT_HARD_SPI) {

1800 (y >= _height) || // " top
1801 ((x2 = (x + w - 1)) < 0) || // " left
1802 ((y2 = (y + h - 1)) < 0))
1803 return; // " bottom
1804
1805 int16_t bx1 = 0, by1 = 0, // Clipped top-left within bitmap
1806 saveW = w; // Save original bitmap width value
1807 if (x < 0) { // Clip left
1808 w += x;

1810 x = 0;
1811 }
1812 if (y < 0) { // Clip top
1813 h += y;
1814 by1 = -y;
1815 y = 0;

1817 if (x2 >= _width)
1818 w = _width - x; // Clip right
1819 if (y2 >= _height)
1820 h = _height - y; // Clip bottom
1821
1822 pcolors += by1 * saveW + bx1; // Offset bitmap ptr to clipped top-left
1823 startWrite();
1824 setAddrWindow(x, y, w, h); // Clipped area
1825 while (h--) { // For each (clipped) scanline...
