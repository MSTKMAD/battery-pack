211bool playSound(int16_t num)
212{
213 static uint16_t index = 0;
214 static bool buzzer_state = C_BUZZER_NOT_BUSSY;
215 static uint16_t local_sound[100];
216 static uint16_t size_sound = 0;
217 if (buzzer_state == C_BUZZER_NOT_BUSSY)
218 {
219 size_sound = getSound(local_sound, num);
220 index = 0;
221 tone(C_PIN_BUZZER, local_sound[index], local_sound[index + 1]);
222 timer_buzzer.set(local_sound[index + 1] + local_sound[index + 2]);
223 buzzer_state = C_BUZZER_BUSSY;
224 }
225 else if (buzzer_state == C_BUZZER_BUSSY)
226 {
227 if (timer_buzzer.poll() != C_TIMER_NOT_EXPIRED)
228 {
229 index += 3;
230 if ((index / 3) >= size_sound)
231 {
232 buzzer_state = C_BUZZER_NOT_BUSSY;
233 //noTone(C_PIN_BUZZER);
234 }
235 else
236 {
237 tone(C_PIN_BUZZER, local_sound[index], local_sound[index + 1]);
238 timer_buzzer.set(local_sound[index + 1] + local_sound[index + 2]);
239 buzzer_state = C_BUZZER_BUSSY;
240 }
241 }
242 }
243 return buzzer_state;
