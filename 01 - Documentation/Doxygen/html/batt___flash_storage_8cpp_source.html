38static inline uint32_t read_unaligned_uint32(const void *data)
39{
40 union {
41 uint32_t u32;
42 uint8_t u8[4];
43 } res;
44 const uint8_t *d = (const uint8_t *)data;
45 res.u8[0] = d[0];
46 res.u8[1] = d[1];
47 res.u8[2] = d[2];
48 res.u8[3] = d[3];
49 return res.u32;
50}

54static void invalidate_CMCC_cache()
55{
56 if (CMCC->SR.bit.CSTS) {
57 CMCC->CTRL.bit.CEN = 0;
58 while (CMCC->SR.bit.CSTS) {}
59 CMCC->MAINT0.bit.INVALL = 1;
60 CMCC->CTRL.bit.CEN = 1;
61 }

65void FlashClass::write(const volatile void *flash_ptr, const void *data, uint32_t size)
66{
67 // Calculate data boundaries
68 size = (size + 3) / 4;
69 volatile uint32_t *dst_addr = (volatile uint32_t *)flash_ptr;
70 const uint8_t *src_addr = (uint8_t *)data;
71
72 // Disable automatic page write
73#if defined(__SAMD51__)
74 NVMCTRL->CTRLA.bit.WMODE = 0;
75 while (NVMCTRL->STATUS.bit.READY == 0) { }
76 // Disable NVMCTRL cache while writing, per SAMD51 errata.
77 bool original_CACHEDIS0 = NVMCTRL->CTRLA.bit.CACHEDIS0;
78 bool original_CACHEDIS1 = NVMCTRL->CTRLA.bit.CACHEDIS1;
79 NVMCTRL->CTRLA.bit.CACHEDIS0 = true;
80 NVMCTRL->CTRLA.bit.CACHEDIS1 = true;
81#else
82 NVMCTRL->CTRLB.bit.MANW = 1;
83#endif
84
85 // Do writes in pages
86 while (size) {
87 // Execute "PBC" Page Buffer Clear
88#if defined(__SAMD51__)
89 NVMCTRL->CTRLB.reg = NVMCTRL_CTRLB_CMDEX_KEY | NVMCTRL_CTRLB_CMD_PBC;
90 while (NVMCTRL->INTFLAG.bit.DONE == 0) { }
91#else
92 NVMCTRL->CTRLA.reg = NVMCTRL_CTRLA_CMDEX_KEY | NVMCTRL_CTRLA_CMD_PBC;
93 while (NVMCTRL->INTFLAG.bit.READY == 0) { }
94#endif
95
96 // Fill page buffer
97 uint32_t i;
98 for (i=0; i<(PAGE_SIZE/4) && size; i++) {
99 *dst_addr = read_unaligned_uint32(src_addr);
100 src_addr += 4;
101 dst_addr++;
102 size--;
103 }
104
105 // Execute "WP" Write Page
106#if defined(__SAMD51__)
107 NVMCTRL->CTRLB.reg = NVMCTRL_CTRLB_CMDEX_KEY | NVMCTRL_CTRLB_CMD_WP;
108 while (NVMCTRL->INTFLAG.bit.DONE == 0) { }
109 invalidate_CMCC_cache();
110 // Restore original NVMCTRL cache settings.
111 NVMCTRL->CTRLA.bit.CACHEDIS0 = original_CACHEDIS0;
112 NVMCTRL->CTRLA.bit.CACHEDIS1 = original_CACHEDIS1;
113#else
114 NVMCTRL->CTRLA.reg = NVMCTRL_CTRLA_CMDEX_KEY | NVMCTRL_CTRLA_CMD_WP;
115 while (NVMCTRL->INTFLAG.bit.READY == 0) { }
116#endif
117 }
118}

120void FlashClass::erase(const volatile void *flash_ptr, uint32_t size)
121{
122 const uint8_t *ptr = (const uint8_t *)flash_ptr;
123 while (size > ROW_SIZE) {
124 erase(ptr);
125 ptr += ROW_SIZE;
126 size -= ROW_SIZE;
127 }
128 erase(ptr);
129}

131void FlashClass::erase(const volatile void *flash_ptr)
132{
133#if defined(__SAMD51__)
134 NVMCTRL->ADDR.reg = ((uint32_t)flash_ptr);
135 NVMCTRL->CTRLB.reg = NVMCTRL_CTRLB_CMDEX_KEY | NVMCTRL_CTRLB_CMD_EB;
136 while (!NVMCTRL->INTFLAG.bit.DONE) { }
137 invalidate_CMCC_cache();
138#else
139 NVMCTRL->ADDR.reg = ((uint32_t)flash_ptr) / 2;
140 NVMCTRL->CTRLA.reg = NVMCTRL_CTRLA_CMDEX_KEY | NVMCTRL_CTRLA_CMD_ER;
141 while (!NVMCTRL->INTFLAG.bit.READY) { }
142#endif
143}
