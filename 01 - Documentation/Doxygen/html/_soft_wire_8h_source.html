109 uint8_t requestFrom(uint8_t address, uint8_t quantity,
110 uint32_t iaddress, uint8_t isize, uint8_t sendStop) {
111 uint8_t localerror = 0;
112 if (isize > 0) {
113 // send internal address; this mode allows sending a repeated start to access
114 // some devices' internal registers. This function is executed by the hardware
115 // TWI module on other processors (for example Due's TWI_IADR and TWI_MMR registers)
116 beginTransmission(address);
117 // the maximum size of internal address is 3 bytes
118 if (isize > 3){
119 isize = 3;
120 }
121 // write internal register address - most significant byte first
122 while (isize-- > 0)
123 write((uint8_t)(iaddress >> (isize*8)));
124 endTransmission(false);
125 }
126 // clamp to buffer length
127 if(quantity > BUFFER_LENGTH){
128 quantity = BUFFER_LENGTH;
129 }
130 if (transmitting)
131 localerror = !i2c_rep_start((address<<1) | I2C_READ);
132 else
133 localerror = !i2c_start((address<<1) | I2C_READ);
134 if (error == 0 && localerror) error = 2;
135 // perform blocking read into buffer
136 for (uint8_t cnt=0; cnt < quantity; cnt++)
137 rxBuffer[cnt] = i2c_read(cnt == quantity-1);
138 // set rx buffer iterator vars
139 rxBufferIndex = 0;
140 rxBufferLength = quantity;
141 if (sendStop) {
142 transmitting = 0;
143 i2c_stop();
144 }
145 return quantity;
146 }
