55void loop()
56{
57 byte error, address;
58 int nDevices;
59
60 Serial.println(F("Scanning I2C bus (7-bit addresses) ..."));
61
62 nDevices = 0;
63 for(address = 1; address < 127; address++ )
64 {
65 // The i2c_scanner uses the return value of
66 // the Write.endTransmisstion to see if
67 // a device did acknowledge to the address.
68 Wire.beginTransmission(address);
69 error = Wire.endTransmission();
70
71 if (error == 0)
72 {
73 Serial.print(F("I2C device found at address 0x"));
74 if (address<16)
75 Serial.print(F("0"));
76 Serial.print(address,HEX);
77 Serial.println(F(" !"));
78
79 nDevices++;
80 }
81 else if (error==4)
82 {
83 Serial.print(F("Unknow error at address 0x"));
84 if (address<16)
85 Serial.print("0");
86 Serial.println(address,HEX);
87 }
88 }
89 if (nDevices == 0)
90 Serial.println("No I2C devices found\n");
91 else
92 Serial.println("done\n");
93
94 delay(5000); // wait 5 seconds for next scan
95}
