31boolean readEEPROM(unsigned long address, uint8_t *byte) {
32 // issue a start condition, send device address and write direction bit
33 if (!i2c_start(EEPROMADDR | I2C_WRITE | (address&0x10000 ? 8 : 0) )) return false;
34
35 // send the address
36 if (!i2c_write((address>>8)&0xFF)) return false;
37 if (!i2c_write(address&0xFF)) return false;
38
39 // issue a repeated start condition, send device address and read direction bit
40 if (!i2c_rep_start(EEPROMADDR | I2C_READ | (address&0x10000 ? 8 : 0) ))return false;
41
42 *byte = i2c_read(true);
43
44 i2c_stop();
45 return true;
46}

49 *burst read

51boolean readBurstEEPROM(unsigned long start, unsigned long stop) {
52 // does not handle the transition from 0x0FFFF to 0x10000
53 // since we only use it for performance evaluation, we do not care!
54 unsigned long addr = start;
55 uint8_t byte;
56 // issue a start condition, send device address and write direction bit
57 if (!i2c_start(EEPROMADDR | I2C_WRITE | (addr&0x10000 ? 8 : 0) )) return false;
58
59 // send the address
60 if (!i2c_write((addr>>8)&0xFF)) return false;
61 if (!i2c_write(addr&0xFF)) return false;
62
63 // issue a repeated start condition, send device address and read direction bit
64 if (!i2c_rep_start(EEPROMADDR | I2C_READ | (addr&0x10000 ? 8 : 0) ))return false;
65 addr++;
66 while (addr++ < stop) byte = i2c_read(false);
67 byte = i2c_read(true);
68 i2c_stop();
69 return true;
70}

99/*
100 * delete eeprom
101 */
102boolean deleteEEPROM(long unsigned from, unsigned long to, uint8_t byte,
103 boolean poll) {
104
105 unsigned long tempto, i;
106 boolean firstpage = true;
107
108 while (from <= to) {
109 tempto = ((from/128)+1)*128-1;
110 if (tempto > to) tempto = to;
111 if (firstpage || !poll) {
112 if (!i2c_start(EEPROMADDR | I2C_WRITE | (from&0x10000 ? 8 : 0)))
113 return false;
114 } else i2c_start_wait(EEPROMADDR | I2C_WRITE | (from&0x10000 ? 8 : 0));
115 // send the address
116 if (!i2c_write((from>>8)&0xFF)) return false;
117 if (!i2c_write(from&0xFF)) return false;
118
119
120 // send data to EEPROM
121 for (i=from; i<=tempto; i++)
122 if (!i2c_write(byte)) return false;
123 // issue a stop condition
124 i2c_stop();
125
126 // wait for ack again
127 if (!poll) delay(6);
128
129 from = tempto+1;
130 firstpage = false;
131 }
132 return true;
133}

196unsigned long parseHex() {
197 unsigned long result = 0L;
198 char inp = '\0';
199 byte val = 0;
200
201 while (inp != '\r' && inp != '#') {
202 while (!Serial.available());
203 inp = Serial.read();
204 if (inp == '\r' || inp == '#') break;
205 if (inp >= 'a' && inp <= 'f')
206 inp = inp -'a' + 'A';
207 if ((inp >= '0' && inp <= '9') ||
208 (inp >= 'A' && inp <= 'F')) {
209 Serial.print(inp);
210 if (inp >= '0' && inp <= '9') val = inp - '0';
211 else val = inp - 'A' + 10;
212 result = result * 16;
213 result = result + val;
214 }
215 }
216 Serial.println();
217 return result;
218}
