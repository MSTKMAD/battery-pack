21#include <RTCZero.h>

23/* Create an rtc object */
24RTCZero rtc;

26/* Change these values to set the current initial time */
27const byte seconds = 0;

29const byte hours = 17;

31/* Change these values to set the current initial date */
32const byte day = 17;

36void setup()
37{
38 pinMode(LED_BUILTIN, OUTPUT);
39 digitalWrite(LED_BUILTIN, LOW);
40
41 rtc.begin();
42
43 rtc.setTime(hours, minutes, seconds);
44 rtc.setDate(day, month, year);
45
46 rtc.setAlarmTime(17, 00, 10);
47 rtc.enableAlarm(rtc.MATCH_HHMMSS);
48
49 rtc.attachInterrupt(alarmMatch);
50
51 rtc.standbyMode();
52}
53
54void loop()
55{
56 rtc.standbyMode(); // Sleep until next alarm match
57}

59void alarmMatch()
