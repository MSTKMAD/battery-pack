50// Configures the design capacity of the connected battery.
51bool BQ27441::setCapacity(uint16_t capacity)
52{
53 // Write to STATE subclass (82) of BQ27441 extended memory.
54 // Offset 0x0A (10)
55 // Design capacity is a 2-byte piece of data - MSB first
56 // Unit: mAh
57 uint8_t capMSB = capacity >> 8;
58 uint8_t capLSB = capacity & 0x00FF;
59 uint8_t capacityData[2] = {capMSB, capLSB};
60 return writeExtendedData(BQ27441_ID_STATE, 10, capacityData, 2);
61}

78{
79 // Write to STATE subclass (82) of BQ27441 extended memory.
80 // Offset 0x0F (16)

238// Set GPOUT polarity to active-high or active-low
239bool BQ27441::setGPOUTPolarity(bool activeHigh)
240{
241 uint16_t oldOpConfig = opConfig();
242
243 // Check to see if we need to update opConfig:
244 if ((activeHigh && (oldOpConfig & BQ27441_OPCONFIG_GPIOPOL)) ||
245 (!activeHigh && !(oldOpConfig & BQ27441_OPCONFIG_GPIOPOL)))
246 return true;
247
248 uint16_t newOpConfig = oldOpConfig;
249 if (activeHigh)
250 newOpConfig |= BQ27441_OPCONFIG_GPIOPOL;
251 else
252 newOpConfig &= ~(BQ27441_OPCONFIG_GPIOPOL);
253
254 return writeOpConfig(newOpConfig);
255}

265// Set GPOUT function to BAT_LOW or SOC_INT
266bool BQ27441::setGPOUTFunction(gpout_function function)
267{
268 uint16_t oldOpConfig = opConfig();
269
270 // Check to see if we need to update opConfig:
271 if ((function && (oldOpConfig & BQ27441_OPCONFIG_BATLOWEN)) ||

273 return true;
274
275 // Modify BATLOWN_EN bit of opConfig:

277 if (function)

281
282 // Write new opConfig
283 return writeOpConfig(newOpConfig);

406// Enter configuration mode - set userControl if calling from an Arduino sketch
407// and you want control over when to exitConfig
408bool BQ27441::enterConfig(bool userControl)
409{
410 if (userControl) _userConfigControl = true;
411
412 if (sealed())
413 {
414 _sealFlag = true;
415 unseal(); // Must be unsealed before making changes
416 }
417
418 if (executeControlWord(BQ27441_CONTROL_SET_CFGUPDATE))
419 {
420 int16_t timeout = BQ72441_I2C_TIMEOUT;
421 while ((timeout--) && (!(flags() & BQ27441_FLAG_CFGUPMODE)))
422 delay(1);
423
424 if (timeout > 0)
425 return true;
426 }
427
428 return false;
429}
430
431// Exit configuration mode with the option to perform a resimulation
432bool BQ27441::exitConfig(bool resim)
433{
434 // There are two methods for exiting config mode:
435 // 1. Execute the EXIT_CFGUPDATE command
436 // 2. Execute the SOFT_RESET command
437 // EXIT_CFGUPDATE exits config mode _without_ an OCV (open-circuit voltage)
438 // measurement, and without resimulating to update unfiltered-SoC and SoC.
439 // If a new OCV measurement or resimulation is desired, SOFT_RESET or
440 // EXIT_RESIM should be used to exit config mode.
441 if (resim)
442 {
443 if (softReset())
444 {
445 int16_t timeout = BQ72441_I2C_TIMEOUT;
446 while ((timeout--) && ((flags() & BQ27441_FLAG_CFGUPMODE)))
447 delay(1);
448 if (timeout > 0)
449 {
450 if (_sealFlag) seal(); // Seal back up if we IC was sealed coming in
451 return true;
452 }
453 }
454 return false;
455 }
456 else
457 {
458 return executeControlWord(BQ27441_CONTROL_EXIT_CFGUPDATE);
459 }
460}

532// Read a 16-bit subcommand() from the BQ27441-G1A's control()
533uint16_t BQ27441::readControlWord(uint16_t function)
534{
535 uint8_t subCommandMSB = (function >> 8);
536 uint8_t subCommandLSB = (function & 0x00FF);
537 uint8_t command[2] = {subCommandLSB, subCommandMSB};
538 uint8_t data[2] = {0, 0};
539
540 i2cWriteBytes((uint8_t) 0, command, 2);
541
542 if (i2cReadBytes((uint8_t) 0, data, 2))
543 {
544 return ((uint16_t)data[1] << 8) | data[0];
545 }
546
547 return false;
548}

611// Read all 32 bytes of the loaded extended data and compute a
612// checksum based on the values.
613uint8_t BQ27441::computeBlockChecksum(void)
614{
615 uint8_t data[32];
616 i2cReadBytes(BQ27441_EXTENDED_BLOCKDATA, data, 32);
617
618 uint8_t csum = 0;
619 for (int i=0; i<32; i++)
620 {
621 csum += data[i];
622 }
623 csum = 255 - csum;
624
625 return csum;
626}

634// Read a byte from extended data specifying a class ID and position offset
635uint8_t BQ27441::readExtendedData(uint8_t classID, uint8_t offset)
636{
637 uint8_t retData = 0;
638 if (!_userConfigControl) enterConfig(false);
639
640 if (!blockDataControl()) // // enable block data memory control
641 return false; // Return false if enable fails
642 if (!blockDataClass(classID)) // Write class ID using DataBlockClass()
643 return false;
644
645 blockDataOffset(offset / 32); // Write 32-bit block offset (usually 0)
646
647 computeBlockChecksum(); // Compute checksum going in
648 uint8_t oldCsum = blockDataChecksum();
649 /*for (int i=0; i<32; i++)
650 Serial.print(String(readBlockData(i)) + " ");*/
651 retData = readBlockData(offset % 32); // Read from offset (limit to 0-31)
652
653 if (!_userConfigControl) exitConfig();
654
655 return retData;
656}

658// Write a specified number of bytes to extended data specifying a
659// class ID, position offset.
660bool BQ27441::writeExtendedData(uint8_t classID, uint8_t offset, uint8_t * data, uint8_t len)
661{
662 if (len > 32)
663 return false;
664
665 if (!_userConfigControl) enterConfig(false);
666
667 if (!blockDataControl()) // // enable block data memory control
668 return false; // Return false if enable fails
669 if (!blockDataClass(classID)) // Write class ID using DataBlockClass()
670 return false;
671
672 blockDataOffset(offset / 32); // Write 32-bit block offset (usually 0)
673 computeBlockChecksum(); // Compute checksum going in
674 uint8_t oldCsum = blockDataChecksum();
675
676 // Write data bytes:
677 for (int i = 0; i < len; i++)
678 {
679 // Write to offset, mod 32 if offset is greater than 32
680 // The blockDataOffset above sets the 32-bit block
681 writeBlockData((offset % 32) + i, data[i]);
682 }
683
684 // Write new checksum using BlockDataChecksum (0x60)
685 uint8_t newCsum = computeBlockChecksum(); // Compute the new checksum
686 writeBlockChecksum(newCsum);
687
688 if (!_userConfigControl) exitConfig();
689
690 return true;
691}

697// Read a specified number of bytes over I2C at a given subAddress
698int16_t BQ27441::i2cReadBytes(uint8_t subAddress, uint8_t * dest, uint8_t count)
699{
700 int16_t timeout = BQ72441_I2C_TIMEOUT;
701 Wire.beginTransmission(_deviceAddress);
702 Wire.write(subAddress);
703 Wire.endTransmission(true);
704
705 Wire.requestFrom(_deviceAddress, count);
706
707 for (int i=0; i<count; i++)
708 {
709 dest[i] = Wire.read();
710 }
711
712 return timeout;
713}
714
715// Write a specified number of bytes over I2C to a given subAddress
716uint16_t BQ27441::i2cWriteBytes(uint8_t subAddress, uint8_t * src, uint8_t count)
717{
718 Wire.beginTransmission(_deviceAddress);
719 Wire.write(subAddress);
720 for (int i=0; i<count; i++)
721 {
722 Wire.write(src[i]);
723 }
724 Wire.endTransmission(true);
725
726 return true;
727}
